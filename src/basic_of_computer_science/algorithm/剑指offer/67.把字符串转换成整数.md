class Solution {
    public int myAtoi(String str) {
        if (str == null || str.length() == 0) return 0;
        char[] strArr = str.toCharArray();
        char[] res = new char[strArr.length];
        // å»é™¤å‰å¯¼ç©ºæ ¼
        int i = 0;
        int j = 0;
        while (i < strArr.length && strArr[i] == ' ') {
            i++;
        }
        // + - å·
        if (i < strArr.length && (strArr[i] == '+' || strArr[i] == '-')) {
            res[j++] = strArr[i++];
        }
        // æ•°å­—
        while (i < strArr.length && strArr[i] >= '0' && strArr[i] <= '9') {
            res[j++] = strArr[i++];
        }
        // char[] è½¬ longï¼Ÿ
        int k = res.length - 1;
        long resLong = 0;
        while (k >= 0 && res[k] == '\0') k--;
        long tmp = 1;
        boolean negative = (res[0] == '-');
        for (; k >= 0; k--) {
            if (res[k] == '+') continue;
            else if (res[k] == '-'){
                resLong *= -1;
                negative = true;
                continue;
            }
            resLong += (res[k] - '0') * tmp;
            tmp *= 10;
            if (!negative && resLong > Integer.MAX_VALUE) return Integer.MAX_VALUE;
            else if (negative && (resLong < Integer.MIN_VALUE || resLong > Integer.MAX_VALUE)) 
                return Integer.MIN_VALUE;
        }
        return (int) resLong;
    }
}

class Solution {
    public int myAtoi(String str) {
        char[] c = str.trim().toCharArray();
        if(c.length == 0) return 0;
        int res = 0, bndry = Integer.MAX_VALUE / 10;
        int i = 1, sign = 1;
        if(c[0] == '-') sign = -1;
        else if(c[0] != '+') i = 0;
        for(int j = i; j < c.length; j++) {
            if(c[j] < '0' || c[j] > '9') break;
            if(res > bndry || res == bndry && c[j] > '7') return sign == 1 ? Integer.MAX_VALUE : Integer.MIN_VALUE;
            res = res * 10 + (c[j] - '0');
        }
        return sign * res;
    }
}
